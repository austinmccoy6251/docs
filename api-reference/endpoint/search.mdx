---
title: '/search'
openapi: 'POST /search'
---

## Header

<ParamField header="Authorization" type="string"  required>
  Bearer authentication header of the form `Bearer <API_KEY>`, where `<API_KEY>` is your organization's API key.
</ParamField>

## Body

<ParamField query="projectId" type="string" required>
  The project ID to perform the search on. This is displayed in the URL when viewing your project on Artemis Search.
</ParamField>

<ParamField query="searchQuery" type="string" required>
  The search query string. For example, "companies that require HIPAA compliance"
</ParamField>

<ParamField query="numberOfBatches" type="integer" minimum="10" maximum="70" required>
  The number of batches of synthetic data to generate for the search. Increasing this may improve search quality but will increase response time.
</ParamField>

<ParamField query="clearFlag" type="boolean" required>
  If set to true, clears any cached results for this query.
</ParamField>

<ParamField query="topK" type="integer" required>
  The maximum number of results to return.
</ParamField>

## Response

<ResponseField name="data" type="array | null">
  <Expandable title="properties">
    <ResponseField name="tag" type="string">
      The tag associated with the matching embedding.
    </ResponseField>
    <ResponseField name="proba" type="number">
      The probability that the result matches the search query, ranging from 0 to 1.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="success" type="boolean">
  Indicates if the search was successful.
</ResponseField>

<ResponseField name="errorMessage" type="string | null">
  An error message if the search was not successful.
</ResponseField>
